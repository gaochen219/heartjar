const OSS = require('ali-oss'); // 从环境变量读取配置 const client = new OSS({ region: process.env.OSS_REGION, accessKeyId: process.env.ACCESS_KEY_ID, accessKeySecret: process.env.ACCESS_KEY_SECRET, bucket: process.env.OSS_BUCKET }); const ROOM_KEY = 'room.json'; const WHISPERS_KEY = 'whispers.json'; const SPARKS_KEY = 'sparks.json'; // 读取 OSS 对象 async function getJSON(key, defaultValue) { try { const result = await client.get(key); return JSON.parse(result.content.toString()); } catch (e) { if (e.code === 'NoSuchKey') return defaultValue; throw e; } } // 写入 OSS 对象 async function putJSON(key, data) { await client.put(key, Buffer.from(JSON.stringify(data))); } exports.handler = async (req, resp, context) => { const cors = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', 'Content-Type': 'application/json' }; // 处理 CORS preflight if (req.method === 'OPTIONS') { resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(''); return; } const path = req.path || '/'; try { // GET /state - 获取房间状态 if (req.method === 'GET' && path === '/state') { const [room, whispers] = await Promise.all([ getJSON(ROOM_KEY, { heartCount: 0 }), getJSON(WHISPERS_KEY, []) ]); resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ heartCount: room.heartCount, whispers })); return; } // POST /heart - 增加爱心 if (req.method === 'POST' && path === '/heart') { const room = await getJSON(ROOM_KEY, { heartCount: 0 }); room.heartCount = (room.heartCount || 0) + 1; await putJSON(ROOM_KEY, room); resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ heartCount: room.heartCount })); return; } // GET /sparks - 获取照片列表 if (req.method === 'GET' && path === '/sparks') { const sparks = await getJSON(SPARKS_KEY, []); resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ sparks })); return; } // POST /spark - 上传照片 if (req.method === 'POST' && path === '/spark') { const body = JSON.parse(req.body.toString()); const sparks = await getJSON(SPARKS_KEY, []); sparks.push({ id: Date.now().toString(), user: body.user, image: body.image, timestamp: Date.now() }); await putJSON(SPARKS_KEY, sparks); resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ success: true })); return; } // POST /whisper - 发送消息 if (req.method === 'POST' && path === '/whisper') { const body = JSON.parse(req.body.toString()); const whispers = await getJSON(WHISPERS_KEY, []); whispers.push({ id: Date.now().toString(), user: body.user, text: body.text, timestamp: Date.now(), read: false }); await putJSON(WHISPERS_KEY, whispers); resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ success: true })); return; } // POST /whisper/read - 标记消息已读 if (req.method === 'POST' && path === '/whisper/read') { const body = JSON.parse(req.body.toString()); const whispers = await getJSON(WHISPERS_KEY, []); const msg = whispers.find(w => w.id === body.id); if (msg) msg.read = true; await putJSON(WHISPERS_KEY, whispers); resp.setStatusCode(200); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ success: true })); return; } // 404 resp.setStatusCode(404); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ error: 'Not found' })); } catch (e) { resp.setStatusCode(500); Object.entries(cors).forEach(([k, v]) => resp.setHeader(k, v)); resp.send(JSON.stringify({ error: e.message })); } };
